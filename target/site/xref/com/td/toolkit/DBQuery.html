26  public class DBQuery {
27  
28      final static Logger logger = LoggerFactory.getLogger(DBQuery.class);
29  
30      private QueryExecutor executor;
31  
32      public DBQuery() {
33          executor = new QueryExecutor();
34      }
35  
36      public static void main(String[] args) {
37          DBQueryuery">DBQuery query = new DBQuery();
38          try {
39              query.run(query.parseArgs(args));
40              query.end();
41          }catch (ParseException e) {
42              System.out.println(e.getMessage());
43              logger.debug( e.getMessage(), e);
44              System.exit(0);
45          }catch (JobFailedException e) {
46              System.out.println(e.getMessage());
47              logger.debug( e.getMessage(), e);
48              System.exit(-1);
49          }catch (InterruptedException e) {
50              System.out.println("Query job is interrupted without succeeding, there could be an error on the Treasure Data server, please contact support with the following  message.\n" + e.getMessage());
51              logger.debug( e.getMessage(), e);
52              System.exit(-1);
53          }
54      }
55  
56      /**
57       * Parsing the arguments passed from CLI
58       *
59       * @param args CLI arguments
60       * @return CommandLine object
61       * @throws ParseException if error happend while parsing
62       */
63      public CommandLine parseArgs(String[] args) throws ParseException {
64          ArgsHandlerHandler">ArgsHandler args_handler = new ArgsHandler(args);
65          // using handler to parse arguments
66          CommandLine cmd = args_handler.parseArgs();
67          return cmd;
68      }
69  
70      /**
71       * Run a command
72       * @param cmd the CommandLine object
73       * @return a map with only one entry which key is the row count and value is records in string
74       * @throws JobFailedException
75       * @throws InterruptedException
76       */
77      public Map run(CommandLine cmd) throws JobFailedException, InterruptedException{
78          // initialize the executor with CommandLine object
79          executor.setCmd(cmd);
80  
81          // execute the query using QueryExecutor object
82          Map result = executor.runSql();
83  
84          // print out the result
85          Map.Entry<Long, String> entry = (Map.Entry<Long, String>) result.entrySet().iterator().next();
86          System.out.println(entry.getValue());
87          return result;
88      }
89  
90      // End up the query process
91      public void end() {
92          executor.terminate();
93      }
94  }
