47      final public static Properties TABLE_COLUMNS = new Properties();
48  
49      public void loadTableColumns() {
50          // using the property file to define the table schema
51          InputStream is = this.getClass().getResourceAsStream("/table_schema.properties");
52          if (is != null ) {
53              try {
54                  TABLE_COLUMNS.load(is);
55              } catch (IOException e) {
56                  logger.debug("table_schema.properties file is not set correctly");
57              } finally {
58                  try {
59                      is.close();
60                  } catch (IOException e) {
61                      logger.error("table_schema.properties file is not closed correctly.", e);
62                  }
63              }
64          }
65      }
